function commonSwitchTab(tabId, contentId, onClass, offClass,onHandle) {
	if(onHandle == 'hover') {
		$("[id^='"+tabId+"_']").each(function(index, dom) {
			var idNum = dom.id.replace(tabId+"_", "");
			$(this).hover(function(){
				$("[id^='"+tabId+"_']").each(function(index_, dom_) {
					var idNum_ = dom_.id.replace(tabId+"_", "");
					if(idNum == idNum_) {
						$("#"+dom_.id).addClass(onClass);
						if(offClass){
						    $("#"+dom_.id).removeClass(offClass);
						}
						$("#"+contentId+"_"+idNum_).show();
					} else {
						$("#"+dom_.id).removeClass(onClass);
						if(offClass){
						    $("#"+dom_.id).addClass(offClass);
						}						
						$("#"+contentId+"_"+idNum_).hide();
					}
				});
			}, function(){
			});
		});
	} else if(onHandle == 'click') {
		$("[id^='"+tabId+"_']").each(function(index, dom) {
			$(this).css("cursor", "pointer");

						}						
						$("#"+contentId+"_"+idNum_).show();
					} else {
						$("#"+dom_.id).removeClass(onClass);
						if(offClass){
						    $("#"+dom_.id).addClass(offClass);
						}									
						$("#"+contentId+"_"+idNum_).hide();
					}
				});
			});
		});
	}
}
